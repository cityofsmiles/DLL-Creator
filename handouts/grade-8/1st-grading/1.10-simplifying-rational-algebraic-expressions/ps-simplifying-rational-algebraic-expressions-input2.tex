\noindent\textbf{Activity 1.10}

%\vspace{0.75ex}

Simplify the following  rational expressions.

\begin{enumerate} 
\begin{multicols}{3}

\item $\displaystyle \frac{x^{2}+3x}{x+3}$
\item $\displaystyle \frac{2x^{2}+8x}{2x}$
\item $\displaystyle \frac{x^{2}-16}{x-4}$
\item $\displaystyle \frac{x^{3}+64}{x^{2}-4x+16}$
\item $\displaystyle \frac{2x^{2}-9x-5}{x-5}$

\item $\displaystyle \frac{2x^{2}+4x}{x+2}$
\item $\displaystyle \frac{4x^{3}-8x^2}{4x^2}$
\item $\displaystyle \frac{4x^{2}-25}{2x-5}$
\item $\displaystyle \frac{8x^{3}-27}{4x^{2}+6x+9}$
\item $\displaystyle \frac{3x^{2}-12x+12}{x-2}$

\end{multicols}
\end{enumerate}
