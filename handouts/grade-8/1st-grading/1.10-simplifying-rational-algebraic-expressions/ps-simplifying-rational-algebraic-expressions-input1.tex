\noindent\textbf{Practice Exercises 1.10}

%\vspace{0.75ex}

Simplify  the following  rational expressions.

\begin{enumerate} 
\begin{multicols}{3}

\item $\displaystyle \frac{4x+16}{2x}$
\item $\displaystyle \frac{x-5}{5-x}$
\item $\displaystyle \frac{3x^{3}+6x^{2}}{3x^{2}}$
\item $\displaystyle \frac{4x^{2}-4x+1}{2x-1}$
\item $\displaystyle \frac{x^{2}+5x+6}{x+3}$
\item $ \dfrac{x^{3} - 27}{x - 3} $
\item $ \dfrac{x^{2} - 25}{x + 5} $
\item $ \dfrac{2x - 6}{x^{2} - 6x + 9} $
\item $ \dfrac{3x^{2} - 12}{2x + 4} $
\item $ \dfrac{3x - 6}{x^{2} - x - 2 } $
\end{multicols}
\end{enumerate}
