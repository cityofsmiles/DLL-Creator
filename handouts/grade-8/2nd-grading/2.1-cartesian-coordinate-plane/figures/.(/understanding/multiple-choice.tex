\newproblem{SPIPoSE-mc-un-1}{%
 Choosing a month from a year, what is the probability of selecting a month with 31 days?
\ifthenelse{\boolean{showanswers}}{}%
{\choice
	{$\dfrac{1}{3}$}
	{$\dfrac{1}{12}$}
	{$\dfrac{7}{12}$}
	{$\dfrac{12}{31}$}
}}{%
\choice
	{$\dfrac{1}{3}$}
	{$\dfrac{1}{12}$}
	{\mbox{\color{red} $\dfrac{7}{12}$}}
	{$\dfrac{12}{31}$}
}
