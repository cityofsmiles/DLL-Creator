\noindent\textbf{Practice Exercises 1.12}

%\vspace{0.75ex}

Solve the following equations using cross-multiplication:

\begin{enumerate}
	\begin{multicols}{3}				
		
	\item \(\dfrac{x}{3} = \dfrac{5}{7}\)
	\item \(\dfrac{2}{x} = \dfrac{8}{12}\)
	\item \(\dfrac{x-1}{4} = \dfrac{7}{10}\)
	\item \(\dfrac{3}{x+2} = \dfrac{6}{11}\)
	\item \(\dfrac{5x}{9} = \dfrac{10}{12}\)
	\item \(\dfrac{x+3}{5} = \dfrac{2}{9}\)
	\item \(\dfrac{4x}{7} = \dfrac{3}{14}\)
	\item \(\dfrac{7}{x-4} = \dfrac{14}{20}\)
	\item \(\dfrac{2x+1}{6} = \dfrac{5}{8}\)
	\item \(\dfrac{9}{x+1} = \dfrac{3}{4}\)
	
	\end{multicols}
\end{enumerate}
