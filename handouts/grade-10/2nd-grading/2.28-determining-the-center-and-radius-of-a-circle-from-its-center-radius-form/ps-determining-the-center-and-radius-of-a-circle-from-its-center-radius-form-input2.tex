    \item $(x - 7)^2 + (y + 3)^2 = 64$
    \item $(x + 2)^2 + (y - 8)^2 = 81$

    \item $(x + 6)^2 + (y - 5)^2 = 121$
    \item $(x - 9)^2 + (y + 7)^2 = 144$
    \item $(x + 8)^2 + (y - 2)^2 = 196$

\end{enumerate}
