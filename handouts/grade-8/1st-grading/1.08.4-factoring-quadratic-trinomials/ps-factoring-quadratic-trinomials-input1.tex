\noindent\textbf{Practice Exercises 1.8.4}

\vspace{0.75ex}

Factor the following polynomials completely. 

\begin{enumerate}
\begin{multicols}{2}	

\item $6x^{2} + x - 2$
\item $3x^{2} + x - 2$
\item $2a^{2} - a - 6 $
\item $4m^{2} + 3mn - n^2$
\item $3a^{2} + ab - 4b^2$
\item $ 4 a^{2} - 6 a b - 4 b^{2} $
\item $ 6 m^{3} - 15 m^{2} - 9 m $
\item $ 3 x^{2} - 8 x y - 3 y^{2} $
\item $ 9 c^{2} d - 15 c d^{2} - 6 d^{3} $
\item $ x^{4} - 8 x^{2} + 15 $			
				
\end{multicols}

\end{enumerate}
