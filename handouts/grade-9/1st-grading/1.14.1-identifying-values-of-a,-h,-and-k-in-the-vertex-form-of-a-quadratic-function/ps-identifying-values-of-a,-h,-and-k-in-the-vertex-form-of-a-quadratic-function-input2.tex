Identify the opening of the parabola, the coordinates of the vertex, and the equation for the axis of symmetry of each quadratic function. 
\begin{multicols}{2}
\begin{enumerate}[label = \color{blue}\arabic*. ]
%1
\item $ y = x^{2} + 8x - 1 $
%2
\item $ y + 1 = 3(x + 2)^{2} $
%3
\item $ (x + 3)^{2} = -y - 6 $
%4
\item $ y + 12x = 4x^{2} - 7 $
%5
\item $ y - 5 = -4(x - 2)^{2} $
\end{enumerate}
\end{multicols}
