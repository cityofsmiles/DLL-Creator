\vspace{1ex}
\noindent\textbf{Practice Exercises 1.11.3}

\vspace{0.75ex}

Find the product of the following rational algebraic expressions. 

\begin{enumerate} 
	\begin{multicols}{2}
	
	%#1
	\item $\displaystyle \frac{10uv^2}{3xy^2} \cdot \frac{6x^2y^2}{5u^2v^2}$ 
	
	%#2
	\item $\displaystyle \frac{a^2-b^2}{2ab} \cdot \frac{a^2}{a-b}$ 
	
	%#3
	\item $\displaystyle \frac{x^2-3x}{x^{2}+3x-10} \cdot \frac{x^2-4}{x^2-x-6}$
	
	%#4
	\item $\displaystyle \frac{x^2+2x+1}{y^{2}-2y+1} \cdot \frac{y^2-1}{x^{2}-1}$
	
	%#5
	\item $\displaystyle \frac{a^2-2ab+b^2}{a^2-1} \cdot \frac{a-1}{a-b}$
	
\end{multicols}
\end{enumerate}
