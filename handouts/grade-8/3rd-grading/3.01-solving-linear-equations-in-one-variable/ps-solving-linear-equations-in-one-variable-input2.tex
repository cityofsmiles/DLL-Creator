\noindent\textbf{Activity 3.1}

\vspace{0.75ex}

Solve the following  linear equations.
\begin{multicols}{2}
\setlist{nolistsep}
\begin{enumerate}[noitemsep, label = \color{blue}\arabic*. ]
    \item $4x - 2 = 10$
    \item $3x + 8 = 20$
    \item $6x + 4 = 3x + 16$
    \item $5x - 7 = x + 11$
    \item $3(x - 4) = 4x + 9$
    \item $5 - 2x = 7$
    \item $9x + 10 = 4x + 3$
    \item $7x - 5 = 2x + 9$
    \item $3(2x - 1) = 9$
    \item $11x + 6 = 2x + 27$
\end{enumerate}
\end{multicols}
