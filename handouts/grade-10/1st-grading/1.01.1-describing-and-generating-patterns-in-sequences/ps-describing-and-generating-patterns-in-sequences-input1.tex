Find the first five terms of each sequence given the nth term.
\begin{multicols}{2}
\begin{enumerate}
    \item $a_n = 3n + 2$
    \item $a_n = 5n - 4$
    \item $a_n = 2^n$
    \item $a_n = \dfrac{n}{n+1}$
    \item $a_n = (-1)^n \cdot n$
    \item $a_n = n^2 + n$
    \item $a_n = \dfrac{3n-1}{2}$
    \item $a_n = 4n^2 - 3$
    \item $a_n = 7 - 2n$
    \item $a_n = \dfrac{5n+3}{n-2}$
    \end{enumerate}
    \end{multicols}
