\noindent\textbf{Activity 3.11}

\vspace{0.75ex}

Find the solutions of the following systems of linear equations graphically.

\begin{enumerate}[label = \arabic*. ]
\begin{multicols}{2}

\item $\left\{
\begin{array}{ccl}
y & = & \displaystyle  \frac{2}{3}x+6 \\
y & = & \displaystyle  -\frac{3}{2}x+6 \\
\end{array}
\right. $
 
\item $\left\{
\begin{array}{ccl}
x+y & = & 7 \\
x-y & = & 1 \\
\end{array}
\right. $
 
\item $\left\{
\begin{array}{ccl}
4x-y & = & 8 \\
3x+2y & = & 6 \\
\end{array}
\right. $
 
\item $\left\{
\begin{array}{ccl}
x+4y & = & 8 \\
x-2y & = & 2 \\
\end{array}
\right. 
$
 
\item $\left\{
\begin{array}{l}
x+y  =  5 \\
 y  = \dfrac{1}{2}x + 2  \\
\end{array}
\right. $

\end{multicols} 
\end{enumerate}
