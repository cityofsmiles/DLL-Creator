\vspace{0.2ex}

Factor the following polynomials completely.
\begin{multicols}{2}
\begin{enumerate}
    \item $x^3 + 3x^2 - x - 3$
    \item $2x^4 - 3x^3 - 2x^2 + 3x$
    \item $x^3 - 8$
    \item $x^4 - 16$
    \item $x^5 - x$
    \item $x^4 + x^3 - x - 1$
    \item $x^6 - 64$
    \item $x^3 + 2x^2 - 4x - 8$
    \item $x^4 + 4x^2 + 4$
    \item $x^5 + x^4 - x - 1$
\end{enumerate}
\end{multicols}
