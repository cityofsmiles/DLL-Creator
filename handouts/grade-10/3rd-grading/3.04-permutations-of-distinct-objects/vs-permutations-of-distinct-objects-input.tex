\noindent  \textbf{Formula for Permutations:}

{\centering $ 
          P(n, r) = \frac{n!}{(n-r)!},
 $\par}         
          where:
          \begin{itemize}
              \item \(n\): Total number of objects.
              \item \(r\): Number of objects selected at a time.
          \end{itemize}
