Graph the following quadratic functions. 
\begin{multicols}{2}
\begin{enumerate}[label = \color{blue}\arabic*. ]
%1
\item $ y = -(x^{2} - 2x ) + 3 $
%2
\item $ 2x^{2} + 8x = 4 - y $
%3
\item $ 4x^{2} - y = 12x - 7 $
%4
\item $ 11 = y - 3x^{2} + 12x $
%5
\item $ 3 = (x - 4)^{2} - y $
\end{enumerate}
\end{multicols}
