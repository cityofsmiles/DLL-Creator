\noindent\textbf{Activity 3.9.1}

\vspace{0.75ex}



For each of the following, you are given two points on a line. Sketch the graph by plotting these points and drawing a line through them.
\begin{multicols}{2}
\setlist{nolistsep}
\begin{enumerate}[noitemsep, label = \color{blue}\arabic*. ]
    \item $(2, 3)$ and $(4, 5)$
    \item $(-1, 0)$ and $(1, 2)$
    \item $(-3, -2)$ and $(1, 2)$
    \item $(0, 1)$ and $(3, -1)$
    \item $(-4, 2)$ and $(1, -3)$
    \item $(3, 1)$ and $(0, -2)$
    \item $(1, 1)$ and $(-2, -4)$
    \item $(2, 2)$ and $(4, 4)$
    \item $(-3, 0)$ and $(1, 4)$
    \item $(0, 5)$ and $(-5, 0)$
\end{enumerate}
\end{multicols}
