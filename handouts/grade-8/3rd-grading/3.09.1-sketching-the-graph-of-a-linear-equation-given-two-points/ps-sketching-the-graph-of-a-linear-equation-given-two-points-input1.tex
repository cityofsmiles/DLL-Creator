\noindent\textbf{Practice Exercises 3.9.1}

\vspace{0.75ex}

For each of the following, you are given two points on a line. Sketch the graph by plotting these points and drawing a line through them.
\begin{multicols}{2}
\setlist{nolistsep}
\begin{enumerate}[noitemsep, label = \color{blue}\arabic*. ]
    \item $(1, 2)$ and $(3, 4)$
    \item $(0, -1)$ and $(2, 1)$
    \item $(-3, 4)$ and $(2, -1)$
    \item $(-1, -2)$ and $(3, 2)$
    \item $(2, 5)$ and $(-1, -4)$
    \item $(4, 0)$ and $(0, 2)$
    \item $(-2, 1)$ and $(1, -3)$
    \item $(0, 0)$ and $(2, 3)$
    \item $(-4, -1)$ and $(2, 3)$
    \item $(5, 3)$ and $(3, 5)$
\end{enumerate}
\end{multicols}
