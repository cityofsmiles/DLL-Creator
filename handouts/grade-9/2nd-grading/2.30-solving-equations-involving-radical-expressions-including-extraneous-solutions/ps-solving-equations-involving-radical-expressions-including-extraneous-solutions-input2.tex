    \item \( \sqrt{3x + 1} = x - 1 \)
    \item \( \sqrt{4x - 8} + 2 = x \)
    \item \( \sqrt{x + 7} = 2x - 1 \)
    \item \( \sqrt{5x + 1} = x + 1 \)
    \item \( \sqrt{6x + 7} = 2x - 1 \)
    \item \( \sqrt{2x + 6} + 1 = x \)
    \item \( \sqrt{7x + 2} = 3x - 2 \)
\end{enumerate}
