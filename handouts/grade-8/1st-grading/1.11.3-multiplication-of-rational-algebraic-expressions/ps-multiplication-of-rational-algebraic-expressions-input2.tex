\noindent\textbf{Activity 1.11.3}

\vspace{0.75ex}

Find the product of the following rational algebraic expressions. 

\begin{enumerate} 
	\begin{multicols}{2}
	
	%#1
	\item $\displaystyle \frac{12mn^2}{6xy^2} \cdot \frac{9x^2y^2}{4m^2n^2}$ 
	
	%#2
	\item $\displaystyle \frac{x^2-y^2}{2xy} \cdot \frac{y^2}{x+y}$ 
	
	%#3
	\item $\displaystyle \frac{x^2-2x}{x^{2}-3x-10} \cdot \frac{x^2-4x-5}{x^2-4}$
	
	%#4
	\item $\displaystyle \frac{x^2+4x+4}{x^{2}+3x+2} \cdot \frac{x^2-1}{x^{2}-4}$
	
	%#5
	\item $\displaystyle \frac{a^2+2ab+b^2}{a^2-b^2} \cdot \frac{a-b}{a+b}$
	
\end{multicols}
\end{enumerate}
