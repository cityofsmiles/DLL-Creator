\noindent\textbf{Practice Exercises 3.1}

\vspace{0.75ex}

Solve the following linear equations.
\begin{multicols}{2}
\setlist{nolistsep}
\begin{enumerate}[noitemsep, label = \color{blue}\arabic*. ]
    \item $3x + 5 = 14$
    \item $2x - 7 = 9$
    \item $5x + 3 = 2x + 12$
    \item $7x - 4 = 2x + 16$
    \item $4(x - 3) = 2x + 8$
    \item $6 - 3x = 12$
    \item $10x + 15 = 5x - 5$
    \item $8x - 9 = 3x + 6$
    \item $2(3x + 1) = 10$
    \item $12x - 7 = 3x + 20$
\end{enumerate}
\end{multicols}
