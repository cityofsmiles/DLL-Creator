\noindent\textbf{Activity 1.12}

%\vspace{0.75ex}

Solve the following equations using cross-multiplication:

\begin{enumerate}
	\begin{multicols}{3}				
		
	\item \(\dfrac{y}{2} = \dfrac{6}{9}\)
	\item \(\dfrac{3}{y} = \dfrac{9}{15}\)
	\item \(\dfrac{y-2}{5} = \dfrac{8}{12}\)
	\item \(\dfrac{4}{y+1} = \dfrac{7}{13}\)
	\item \(\dfrac{6y}{8} = \dfrac{9}{11}\)
	\item \(\dfrac{y+4}{7} = \dfrac{3}{10}\)
	\item \(\dfrac{5y}{9} = \dfrac{4}{15}\)
	\item \(\dfrac{8}{y-3} = \dfrac{16}{24}\)
	\item \(\dfrac{3y+2}{8} = \dfrac{7}{10}\)
	\item \(\dfrac{10}{y+2} = \dfrac{4}{5}\)

\end{multicols}
\end{enumerate}
