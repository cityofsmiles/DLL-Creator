\begin{enumerate} 
	\begin{multicols}{2}
	
	%#1
	\item $\displaystyle \frac{14x^2}{20y^2} \div \frac{56x^2}{y}$ 
	%\vspce
	%#2
	\item $\displaystyle \frac{4a-4b}{30a^2} \div \frac{a-b}{9a}$ 
	%\vspce
	%#3
	\item $\displaystyle \frac{x^2-4}{x^2+4x+4} \div \frac{x^2-x-2}{x^2+3x+2}$
	%\vspce
	%#4
	\item $\displaystyle \frac{x^2-2x-3}{x^{2}-3x} \div \frac{x^2-4}{x^{2}+2x}$
	%\vspce
	%#5
	\item $\displaystyle \frac{x^2-4}{x^2+2x} \div \frac{x^2+x-6}{2x+4}$
	
\end{multicols}
\end{enumerate}
