Factor the following polynomials completely. 

\begin{enumerate}
\begin{multicols}{2}
\item $36x^{2} - 64$
\item $16x^{4} - 49y^{2}z^{2}$
\item $4a^{2} - b^{6}$
\item $81m^{4}n^{2} - 9z^{6}$
\item $a^{4} - 16b^{2}$
\item $16m^{8} - 81b^{4}$
\item $c^{4} - 1$
\item $x^{4}y^2 - 36z^{6}$
\item $x^4y^{2} - 49$
\item $16m^{4} - 64$
\end{multicols} 
\end{enumerate}
