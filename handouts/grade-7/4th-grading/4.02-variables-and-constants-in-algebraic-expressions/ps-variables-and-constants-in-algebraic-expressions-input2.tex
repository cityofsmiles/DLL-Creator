Identify the variables and constants in the following algebraic expressions.
\begin{multicols}{2}				
\begin{enumerate}
    \item $6x - 3$  
    \item $5y + 4$  
    \item $7a - 4b + 8$  
    \item $n^2 + 6n - 12$  
    \item $qr - 2q + 5r - 3$  
    \item $2x^2 + 3x - 4$  
    \item $j + 6k - 8$  
    \item $\frac{3}{4}x - y + 2z - 5$  
    \item $8a^2 + b^2 - 3c + 6$  
    \item $y^3 + 5y - 3x + 12$  
\end{enumerate}
  
\end{multicols}
