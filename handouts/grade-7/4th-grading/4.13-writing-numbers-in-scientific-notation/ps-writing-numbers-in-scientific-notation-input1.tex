\noindent\textbf{A.} Convert the following numbers to scientific notation.
\begin{multicols}{3}
\begin{enumerate}
    \item \( 2,500,000 \)  
    \item \( 0.00045 \)  
    \item \( 78,000 \)  
    \item \( 0.0000079 \)  
    \item \( 9,600,000 \)  
\end{enumerate}
\end{multicols}

\noindent\textbf{B.} Convert the following scientific notation to standard form.
\begin{multicols}{3}
\begin{enumerate}
    \item \( 7.6 \times 10^5 \)  
    \item \( 9.3 \times 10^{-3} \)  
    \item \( 1.25 \times 10^8 \)  
    \item \( 6.45 \times 10^{-6} \)  
    \item \( 3.4 \times 10^2 \)  
\end{enumerate}
\end{multicols}
