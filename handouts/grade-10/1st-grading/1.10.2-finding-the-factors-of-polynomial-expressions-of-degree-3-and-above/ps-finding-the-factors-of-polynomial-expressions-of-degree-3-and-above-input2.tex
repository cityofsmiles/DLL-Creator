Factor the following polynomials completely.
\begin{multicols}{2}
\begin{enumerate}
    \item $x^3 + 2x^2 - x - 2$
    \item $3x^4 - 5x^3 - 3x^2 + 5x$
    \item $x^3 - 27$
    \item $x^4 - 81$
    \item $x^5 - 2x$
    \item $x^4 + 2x^3 - 2x - 2$
    \item $x^6 - 125$
    \item $x^3 + 3x^2 - 9x - 27$
    \item $x^4 + 9x^2 + 9$
    \item $x^5 + 2x^4 - 2x - 2$
\end{enumerate}
\end{multicols}
