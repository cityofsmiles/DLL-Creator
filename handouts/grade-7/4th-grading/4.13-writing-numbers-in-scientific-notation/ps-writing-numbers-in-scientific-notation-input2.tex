\noindent\textbf{A.} Convert the following numbers to scientific notation.
\begin{multicols}{3}
 \begin{enumerate}
    \item \( 1,200,000 \)  
    \item \( 0.00067 \)

\end{enumerate}
\end{multicols}

    \item \( 3.67 \times 10^7 \)  
    \item \( 4.89 \times 10^{-5} \)  
    \item \( 1.2 \times 10^3 \)  
\end{enumerate}
\end{multicols}
