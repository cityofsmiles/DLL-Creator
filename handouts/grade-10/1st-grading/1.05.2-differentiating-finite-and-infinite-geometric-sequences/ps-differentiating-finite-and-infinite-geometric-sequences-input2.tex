Identify whether the given geometric sequence is finite or infinite.
\begin{multicols}{2}
\begin{enumerate}
    \item 4, 8, 16, 32, ..., 512
    \item 3, 6, 12, 24, ...
    \item 6, -12, 24, -48, ..., -1536
    \item \(2, \dfrac{2}{3}, \dfrac{2}{9}, \dfrac{2}{27}, ...\)
    \item 12, 36, 108, 324, ...
    \item \(200, 100, 50, 25, ..., \dfrac{25}{32}\)
    \item -3, -9, -27, -81, ...
    \item 5, 15, 45, 135, ..., 59049
    \item 9, -18, 36, -72, ...
    \item 2, 6, 18, 54, ..., 1458
\end{enumerate}
\end{multicols}
