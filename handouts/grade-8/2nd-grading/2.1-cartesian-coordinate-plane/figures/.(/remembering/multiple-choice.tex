\newproblem{SPIPoSE-mc-re-1}{%
  A 52-card pack is well shuffled and then one card is drawn from the top of the pack. Determine the probability that it is a red number card. 
\ifthenelse{\boolean{showanswers}}{}%
{\choice
	{$\dfrac{1}{2}$}
	{$\dfrac{1}{3}$}
	{$\dfrac{9}{26}$}
	{$\dfrac{25}{52}$}
}}{%
\choice
	{$\dfrac{1}{2}$}
	{$\dfrac{1}{3}$}
	{\mbox{\color{red} $\dfrac{9}{26}$}}
	{$\dfrac{25}{52}$}
}
